16void
17discretize_curve_segments(
18 const std::vector<RationalFunction<degree, dimension>>& curve_segments,
19 const CurveDiscretizationParameters& curve_disc_params,
20 std::vector<Eigen::Matrix<double, 1, dimension>>& points,
21 std::vector<std::vector<int>>& polylines)
22{
23 points.clear();
24 polylines.clear();
25 for (size_t k = 0; k < curve_segments.size(); ++k) {
26 RationalFunction<degree, dimension> const& curve_segment =
27 curve_segments[k];
28
29 // Write curves
30 int num_samples = curve_disc_params.num_samples;
31 std::vector<Eigen::Matrix<double, 1, dimension>> points_k;
32 curve_segment.sample_points(num_samples, points_k);
33
34 // Build polyline for the given curve
35 std::vector<int> polyline;
36 polyline.resize(points_k.size());
37 for (size_t l = 0; l < points_k.size(); ++l) {
38 polyline[l] = points.size() + l;
39 }
40
41 append(points, points_k);
42 polylines.push_back(polyline);
43 }
44}
