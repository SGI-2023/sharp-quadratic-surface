159template<typename Scalar>
160void
161generate_twelve_split_data_to_monomial_matrices(
162 std::array<Eigen::Matrix<Scalar, 6, 12>, 12>& coefficient_matrices)
163{
164 double patch_coeffs[12][6][12];
165 PS12_patch_coeffs(patch_coeffs);
166 for (size_t i = 0; i < 12; ++i) {
167 for (size_t j = 0; j < 6; ++j) {
168 for (size_t k = 0; k < 12; ++k) {
169 coefficient_matrices[i](j, k) = Scalar(patch_coeffs[i][j][k]);
170 }
171 }
172 }
173}

193template<typename Scalar>
194void
195generate_twelve_split_data_matrix(
196 const std::array<TriangleCornerData<Eigen::Matrix<Scalar, 1, 3>>, 3>&
197 corner_data,
198 const std::array<TriangleMidpointData<Eigen::Matrix<Scalar, 1, 3>>, 3>&
199 midpoint_data,
200 Eigen::Matrix<Scalar, 12, 3>& twelve_split_data)
201{
202 // Vertex positions
203 twelve_split_data.row(0) = corner_data[0].function_value;
204 twelve_split_data.row(1) = corner_data[1].function_value;
205 twelve_split_data.row(2) = corner_data[2].function_value;
206
207 // Vertex gradients
208 twelve_split_data.row(3) = corner_data[0].first_edge_derivative;
209 twelve_split_data.row(4) = corner_data[0].second_edge_derivative;
210 twelve_split_data.row(5) = corner_data[1].second_edge_derivative;
211 twelve_split_data.row(6) = corner_data[1].first_edge_derivative;
212 twelve_split_data.row(7) = corner_data[2].first_edge_derivative;
213 twelve_split_data.row(8) = corner_data[2].second_edge_derivative;
214
215 // Edge midpoint normals in order ij, jk, ki
216 twelve_split_data.row(9) = midpoint_data[2].normal_derivative;
217 twelve_split_data.row(10) = midpoint_data[0].normal_derivative;
218 twelve_split_data.row(11) = midpoint_data[1].normal_derivative;
219}

228template<typename Scalar>
229void
230generate_twelve_split_spline_patch_surface_mapping(
231 const std::array<TriangleCornerData<Eigen::Matrix<Scalar, 1, 3>>, 3>&
232 corner_data,
233 const std::array<TriangleMidpointData<Eigen::Matrix<Scalar, 1, 3>>, 3>&
234 midpoint_data,
235 std::array<Eigen::Matrix<Scalar, 6, 3>, 12>& surface_mappings)
236{
237 // Generate matrices to go from the position data to surface coefficients
238 std::array<Eigen::Matrix<Scalar, 6, 12>, 12> coefficient_matrices;
239 generate_twelve_split_data_to_monomial_matrices<Scalar>(coefficient_matrices);
240
241 // Combine position data into a matrix
242 Eigen::Matrix<Scalar, 12, 3> twelve_split_data;
243 generate_twelve_split_data_matrix<Scalar>(
244 corner_data, midpoint_data, twelve_split_data);
245
246 // Build the surface mappings
247 for (size_t i = 0; i < 12; ++i) {
248 surface_mappings[i] = coefficient_matrices[i] * twelve_split_data;
249 }
