238template<typename Gradient, typename Hessian>
239void
240update_energy_quadratic(const double& local_energy,
241 const Gradient& local_derivatives,
242 const Hessian& local_hessian,
243 const std::vector<int>& local_to_global_map,
244 double& energy,
245 VectorXr& derivatives,
246 std::vector<Eigen::Triplet<double>>& hessian_entries)
247{
248 spdlog::trace("Adding local face energy {}", local_energy);
249 SPDLOG_TRACE("Local to global map: {}",
250 formatted_vector(local_to_global_map, ", "));
251 // Update energy
252 energy += local_energy;
253
254 // Update derivatives
255 int num_local_indices = local_to_global_map.size();
256 for (int local_index = 0; local_index < num_local_indices; ++local_index) {
257 int global_index = local_to_global_map[local_index];
258 if (global_index < 0)
259 continue; // Skip fixed variables with no global index
260 derivatives[global_index] += local_derivatives[local_index];
261 }
262
263 // Update hessian entries
264 for (int local_index_i = 0; local_index_i < num_local_indices;
265 ++local_index_i) {
266 // Get global row index, skipping fixed variables with no global index
267 int global_index_i = local_to_global_map[local_index_i];
268 if (global_index_i < 0)
269 continue;
270
271 for (int local_index_j = 0; local_index_j < num_local_indices;
272 ++local_index_j) {
273 // Get global column index, skipping fixed variables with no global index
274 int global_index_j = local_to_global_map[local_index_j];
275 if (global_index_j < 0)
276 continue;
277
278 // Get Hessian entry value
279 double hessian_value = local_hessian(local_index_i, local_index_j);
280
281 // Assemble global Hessian entry
282 hessian_entries.push_back(
283 Eigen::Triplet<double>(global_index_i, global_index_j, hessian_value));
284 }
285 }
286}
