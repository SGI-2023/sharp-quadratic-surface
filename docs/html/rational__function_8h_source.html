103 void compute_derivative(
104 RationalFunction<2 * degree, dimension>& derivative) const
105 {
106 // Compute the derivatives of the numerator and denominator polynomials
107 spdlog::trace("Taking derivative of rational function");
108 spdlog::trace("Numerator:\n{}", m_numerator_coeffs);
109 spdlog::trace("Denominator:\n{}", m_denominator_coeffs);
110 Eigen::Matrix<double, degree, dimension> numerator_deriv_coeffs;
111 compute_polynomial_mapping_derivative<degree, dimension>(
112 m_numerator_coeffs, numerator_deriv_coeffs);
113 Eigen::Matrix<double, degree, 1> denominator_deriv_coeffs;
114 compute_polynomial_mapping_derivative<degree, 1>(m_denominator_coeffs,
115 denominator_deriv_coeffs);
116 spdlog::trace("Numerator derivative:\n{}", numerator_deriv_coeffs);
117 spdlog::trace("Denominator derivative:\n{}", denominator_deriv_coeffs);
118
119 // FIXME 0 degree case
120
121 // Compute the derivative numerator and denominator from the quotient rule
122 Eigen::Matrix<double, 2 * degree, dimension> term_0, term_1;
123 compute_polynomial_mapping_scalar_product<degree, degree - 1, dimension>(
124 m_denominator_coeffs, numerator_deriv_coeffs, term_0);
125 compute_polynomial_mapping_scalar_product<degree - 1, degree, dimension>(
126 denominator_deriv_coeffs, m_numerator_coeffs, term_1);
127 spdlog::trace("First term:\n{}", term_0);
128 spdlog::trace("Second term:\n{}", term_1);
129 Eigen::Matrix<double, 2 * degree + 1, dimension> num_coeffs;
130 num_coeffs.setZero();
131 num_coeffs.block(0, 0, 2 * degree, dimension) = term_0 - term_1;
132 Eigen::Matrix<double, 2 * degree + 1, 1> denom_coeffs;
133 compute_polynomial_mapping_product<degree, degree, 1>(
134 m_denominator_coeffs, m_denominator_coeffs, denom_coeffs);
135
136 // Build the derivative
137 derivative = RationalFunction<2 * degree, dimension>(
138 num_coeffs, denom_coeffs, m_domain);
139 }

165 void split_at_knot(double knot,
166 RationalFunction<degree, dimension>& lower_segment,
167 RationalFunction<degree, dimension>& upper_segment)
168 {
169 // Build lower segment
170 double t0 = m_domain.get_lower_bound();
171 assert(t0 <= knot);
172 Interval lower_domain(t0, knot);
173 lower_segment = RationalFunction<degree, dimension>(
174 m_numerator_coeffs, m_denominator_coeffs, lower_domain);
175
176 // Build upper segment
177 double t1 = m_domain.get_upper_bound();
178 assert(knot <= t1);
179 Interval upper_domain(knot, t1);
180 upper_segment = RationalFunction<degree, dimension>(
181 m_numerator_coeffs, m_denominator_coeffs, upper_domain);
182 }

188 void sample_points(
189 int num_points,
190 std::vector<Eigen::Matrix<double, 1, dimension>>& points) const
191 {
192 // Get sample of the domain
193 std::vector<double> t_samples = m_domain.sample_points(num_points);
194
195 // Evaluate the function at the sampled domain points
196 points.resize(num_points);
197 for (int i = 0; i < num_points; ++i) {
198 evaluate(t_samples[i], points[i]);
199 }

219 Eigen::Matrix<double, 1, dimension> mid_point() const
220 {
221 // Return the default constructor point if the domain is not bounded below
222 if (!m_domain.is_bounded_below())
223 return Eigen::Matrix<double, 1, dimension>();
224 if (!m_domain.is_bounded_above())
225 return Eigen::Matrix<double, 1, dimension>();
226
227 double t0 = m_domain.get_lower_bound();
228 double t1 = m_domain.get_upper_bound();
229 return evaluate((t0 + t1) / 2.0);
230 }

249 void evaluate(double t, Eigen::Matrix<double, 1, dimension>& point) const
250 {
251 Eigen::Matrix<double, 1, dimension> Pt;
252 Eigen::Matrix<double, 1, 1> Qt;
253 evaluate_polynomial_mapping<degree, dimension>(m_numerator_coeffs, t, Pt);
254 evaluate_polynomial_mapping<degree, 1>(m_denominator_coeffs, t, Qt);
255
256 point = Pt / Qt[0];
257 }

263 void evaluate_normalized_coordinate(
264 double t,
265 Eigen::Matrix<double, 1, dimension>& point) const
266 {
267 // Check if domain is bounded
268 if (!domain().is_bounded_below())
269 return;
270 if (!domain().is_bounded_above())
271 return;
272
273 // Linearly interpolate the coordinate
274 double t0 = domain().get_lower_bound();
275 double t1 = domain().get_upper_bound();
276 double s = interval_lerp(0.0, 1.0, t0, t1, t);
277
278 // Evaluate at the given domain coordinate
279 evaluate(s, point);
280 }

301 void discretize(const CurveDiscretizationParameters& curve_disc_params,
302 std::vector<Eigen::Matrix<double, 1, dimension>>& points,
303 std::vector<int>& polyline) const
304 {
305 // Write curves
306 int num_samples = curve_disc_params.num_samples;
307 sample_points(num_samples, points);
308
309 // Build polyline for the given curve
310 polyline.resize(points.size());
311 for (size_t l = 0; l < points.size(); ++l) {
312 polyline[l] = l;
313 }

321 void add_curve_to_viewer(
322 std::string curve_name = "rational_function_curve") const
323 {
324 if (dimension != 3) {

333 discretize(curve_disc_params, points, polyline);
334
335 // Add curve mesh
336 MatrixXr points_mat = convert_nested_vector_to_matrix(points);
337 std::vector<std::vector<int>> polylines = { polyline };
338 std::vector<std::array<int, 2>> edges =
339 convert_polylines_to_edges(polylines);
340 polyscope::init();
341 polyscope::registerCurveNetwork(curve_name, points_mat, edges);
342 polyscope::getCurveNetwork(curve_name)->setRadius(0.0025);
343 }

354 std::vector<Eigen::Matrix<double, 1, dimension>> finite_difference_derivative(
355 double t,
356 double h = 1e-3) const
357 {
358 std::vector<Eigen::Matrix<double, 1, dimension>> F_plus, F_minus;
359 evaluate(t + h, F_plus);
360 evaluate(t - h, F_minus);
361 return (F_plus - F_minus) / (2 * h);
362 }
