81 template<size_t dimension>
82 void pullback_quadratic_function(
83 const Eigen::Matrix<double, 6, dimension>& F_coeffs,
84 RationalFunction<4, dimension>& pullback_function) const
85 {
86 SPDLOG_TRACE("Pulling back conic by quadratic function {}",
87 formatted_bivariate_quadratic_mapping(F_coeffs));
88
89 // Separate the individual polynomial coefficients from the rational
90 // function
91 Matrix3x2r P_coeffs = get_numerators();
92 Eigen::Matrix<double, 3, 1> u_coeffs = P_coeffs.col(0);
93 Eigen::Matrix<double, 3, 1> v_coeffs = P_coeffs.col(1);
94 Eigen::Matrix<double, 3, 1> Q_coeffs = get_denominator();
95 SPDLOG_TRACE("u function before pullback: ({})/({})",
96 formatted_polynomial(u_coeffs),
97 formatted_polynomial(Q_coeffs));
98 SPDLOG_TRACE("v function before pullback: ({})/({})",
99 formatted_polynomial(v_coeffs),
100 formatted_polynomial(Q_coeffs));
101
102 // Compute (homogenized) polynomial coefficients for the quadratic terms
103 Eigen::Matrix<double, 5, 1> QQ_coeffs, Qu_coeffs, Qv_coeffs, uv_coeffs,
104 uu_coeffs, vv_coeffs;
105 compute_polynomial_mapping_product<2, 2, 1>(Q_coeffs, Q_coeffs, QQ_coeffs);
106 compute_polynomial_mapping_product<2, 2, 1>(Q_coeffs, u_coeffs, Qu_coeffs);
107 compute_polynomial_mapping_product<2, 2, 1>(Q_coeffs, v_coeffs, Qv_coeffs);
108 compute_polynomial_mapping_product<2, 2, 1>(u_coeffs, v_coeffs, uv_coeffs);
109 compute_polynomial_mapping_product<2, 2, 1>(u_coeffs, u_coeffs, uu_coeffs);
110 compute_polynomial_mapping_product<2, 2, 1>(v_coeffs, v_coeffs, vv_coeffs);
111
112 // Combine quadratic monomial functions into a matrix
113 Eigen::Matrix<double, 5, 6> monomial_coeffs;
114 monomial_coeffs.setZero();
115 monomial_coeffs.col(0).head(QQ_coeffs.size()) = QQ_coeffs;
116 monomial_coeffs.col(1).head(Qu_coeffs.size()) = Qu_coeffs;
117 monomial_coeffs.col(2).head(Qv_coeffs.size()) = Qv_coeffs;
118 monomial_coeffs.col(3).head(uv_coeffs.size()) = uv_coeffs;
119 monomial_coeffs.col(4).head(uu_coeffs.size()) = uu_coeffs;
120 monomial_coeffs.col(5).head(vv_coeffs.size()) = vv_coeffs;
121 spdlog::trace("Monomial coefficient matrix:\n{}", monomial_coeffs);
122
123 // Compute the pulled back rational function numerator
124 spdlog::trace("Quadratic coefficient matrix:\n{}", F_coeffs);
125 Eigen::Matrix<double, 5, dimension> pullback_coeffs =
126 monomial_coeffs * F_coeffs;
127 SPDLOG_TRACE("Pullback numerator: {}",
128 formatted_polynomial(pullback_coeffs));
129 SPDLOG_TRACE("Pullback denominator: {}", formatted_polynomial(QQ_coeffs));
130
131 pullback_function =
132 RationalFunction<4, dimension>(pullback_coeffs, QQ_coeffs, domain());
133 }
