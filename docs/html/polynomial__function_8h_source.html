26template<int degree>
27void
28generate_monomials(double t, Eigen::Ref<Eigen::Matrix<double, 1, degree + 1>> T)
29{
30 T[0] = 1.0;
31 for (int i = 1; i <= degree; ++i) {
32 T[i] = t * T[i - 1];
33 }

83template<int first_degree, int second_degree, int dimension>
84void
85compute_polynomial_mapping_product(
86 const Eigen::Ref<const Eigen::Matrix<double, first_degree + 1, dimension>>&
87 first_polynomial_coeffs,
88 const Eigen::Ref<const Eigen::Matrix<double, second_degree + 1, dimension>>&
89 second_polynomial_coeffs,
90 Eigen::Ref<Eigen::Matrix<double, first_degree + second_degree + 1, dimension>>
91 product_polynomial_coeffs)
92{
93 // Compute the new polynomial coefficients by convolution
94 product_polynomial_coeffs.setZero();
95 for (int i = 0; i <= first_degree; ++i) {
96 for (int j = 0; j <= second_degree; ++j) {
97 for (int k = 0; k < dimension; ++k) {
98 product_polynomial_coeffs(i + j, k) +=
99 first_polynomial_coeffs(i, k) * second_polynomial_coeffs(j, k);
100 }
101 }
102 }
103}

146template<int first_degree, int second_degree>
147void
148compute_polynomial_mapping_cross_product(
149 const Eigen::Ref<const Eigen::Matrix<double, first_degree + 1, 3>>&
150 first_polynomial_coeffs,
151 const Eigen::Ref<const Eigen::Matrix<double, second_degree + 1, 3>>&
152 second_polynomial_coeffs,
153 Eigen::Ref<Eigen::Matrix<double, first_degree + second_degree + 1, 3>>
154 product_polynomial_coeffs)
155{
156 // Compute cross product terms
157 Eigen::Matrix<double, first_degree + second_degree + 1, 1> A0B1, A0B2, A1B0,
158 A1B2, A2B0, A2B1;
159 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
160 first_polynomial_coeffs.col(0), second_polynomial_coeffs.col(1), A0B1);
161 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
162 first_polynomial_coeffs.col(0), second_polynomial_coeffs.col(2), A0B2);
163 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
164 first_polynomial_coeffs.col(1), second_polynomial_coeffs.col(0), A1B0);
165 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
166 first_polynomial_coeffs.col(1), second_polynomial_coeffs.col(2), A1B2);
167 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
168 first_polynomial_coeffs.col(2), second_polynomial_coeffs.col(0), A2B0);
169 compute_polynomial_mapping_product<first_degree, second_degree, 1>(
170 first_polynomial_coeffs.col(2), second_polynomial_coeffs.col(1), A2B1);
171
172 // Assemble the cross product from the terms
173 product_polynomial_coeffs.col(0) = A1B2 - A2B1;
174 product_polynomial_coeffs.col(1) = A2B0 - A0B2;
175 product_polynomial_coeffs.col(2) = A0B1 - A1B0;
176}

188template<int first_degree, int second_degree, int dimension>
189void
190compute_polynomial_mapping_dot_product(
191 const Eigen::Ref<const Eigen::Matrix<double, first_degree + 1, dimension>>&
192 first_polynomial_coeffs,
193 const Eigen::Ref<const Eigen::Matrix<double, second_degree + 1, dimension>>&
194 second_polynomial_coeffs,
195 Eigen::Ref<Eigen::Matrix<double, first_degree + second_degree + 1, 1>>
196 product_polynomial_coeffs)
197{
198 product_polynomial_coeffs.setZero();
199 for (int i = 0; i < dimension; ++i) {
200 Eigen::Matrix<double, first_degree + second_degree + 1, 1>
201 product_polynomial_term_coeffs;
202 compute_polynomial_mapping_product<first_degree, second_degree, dimension>(
203 first_polynomial_coeffs.col(i),
204 second_polynomial_coeffs.col(i),
205 product_polynomial_term_coeffs);
206 product_polynomial_coeffs += product_polynomial_term_coeffs;
207 }
208}

274template<int degree, int dimension>
275std::string
276formatted_polynomial(
277 const Eigen::Ref<const Eigen::Matrix<double, degree + 1, dimension>>&
278 polynomial_coeffs,
279 int precision = 16)
280{
281 std::stringstream polynomial_string;
282
283 // Handle trivial case
284 if (polynomial_coeffs.cols() == 0)
285 return "";
286
287 for (int i = 0; i < polynomial_coeffs.cols(); ++i) {
288 polynomial_string << std::fixed << std::setprecision(precision)
289 << polynomial_coeffs(0, i);
290 for (int j = 1; j < polynomial_coeffs.rows(); ++j) {
291 std::string monomial_string = formatted_monomial("t", j);
292 polynomial_string << formatted_term(
293 polynomial_coeffs(j, i), monomial_string, precision);
294 }
295 polynomial_string << "\n";
296 }
297
298 return polynomial_string.str();
299}

310template<typename VariableType>
311VariableType
312substitute_polynomial(const VectorXr& A_coeffs, VariableType t)
313{
314 if (A_coeffs.size() == 0)
315 return VariableType();
316
317 // Initialize polynomial evaluation with the constant term
318 VariableType A = VariableType(A_coeffs[0]);
319
320 // Add the higher order terms
321 int degree = A_coeffs.size() - 1;
322 VariableType power_of_t = t;
323 for (int i = 1; i <= degree; ++i) {
324 A = A + VariableType(A_coeffs[i]) * power_of_t;
325 power_of_t = power_of_t * t;
326 }
327
328 return A;
329}
