13class QuadraticPositionFunction{
14public:
15 QuadraticPositionFunction(
16 double _uv_coeff,
17 double _uu_coeff,
18 double _vv_coeff
19 ) {
20 uv_coeff = _uv_coeff;
21 uu_coeff = _uu_coeff;
22 vv_coeff = _vv_coeff;
23 }
24
25 SpatialVector operator()(
26 double u,
27 double v
28 ) {
29 SpatialVector point;
30 point(0) = u;
31 point(1) = v;
32 point(2) = uv_coeff * u * v + uu_coeff * u * u + vv_coeff * v * v;
33
34 return point;
35 }
36
37 double uv_coeff;
38 double uu_coeff;
39 double vv_coeff;
40};

43class QuadraticGradientFunction{
44public:
45 QuadraticGradientFunction(
46 double _uv_coeff,
47 double _uu_coeff,
48 double _vv_coeff
49 ) {
50 uv_coeff = _uv_coeff;
51 uu_coeff = _uu_coeff;
52 vv_coeff = _vv_coeff;
53 }
54
55 Matrix2x3r operator()(
56 double u,
57 double v
58 ) {
59 MatrixXr gradient(2, 3);
60
61 // Generate derivative with respect to u
62 gradient(0, 0) = 1.0;
63 gradient(0, 1) = 0.0;
64 gradient(0, 2) = uv_coeff * v + 2 * uu_coeff * u;
65
66 // Generate derivative with respect to v
67 gradient(1, 0) = 0.0;
68 gradient(1, 1) = 1.0;
69 gradient(1, 2) = uv_coeff * u + 2 * vv_coeff * v;
70
71 return gradient;
72 }
73
74 double uv_coeff;
75 double uu_coeff;
76 double vv_coeff;
77};

114class TorusGradientFunction {
115public:
116 TorusGradientFunction(
117 double _major_radius,
118 double _minor_radius,
119 int _resolution,
120 double _angle_offset
121 ) {
122 R = _major_radius;
123 r = _minor_radius;
124 resolution = _resolution;
125 angle_offset = _angle_offset;
126 }
127
128 Matrix2x3r operator()(
129 double u,
130 double v
131 ) {
132 Matrix2x3r gradient(2, 3);
133
134 // Compute angle derivatives
135 double theta = generate_angle(u, resolution, angle_offset);
136 double phi = generate_angle(v, resolution, angle_offset);
137 double dtheta_du = generate_angle_derivative(resolution);
138 double dphi_dv = generate_angle_derivative(resolution);
139
140 // Generate derivative with respect to u
141 gradient(0, 0) = -r * std::sin(theta) * std::cos(phi) * dtheta_du;
142 gradient(0, 1) = -r * std::sin(theta) * std::sin(phi) * dtheta_du;
143 gradient(0, 2) = r * std::cos(theta) * dtheta_du;
144
145 // Generate derivative with respect to v
146 gradient(1, 0) = -(R + r * std::cos(theta)) * std::sin(phi) * dphi_dv;
147 gradient(1, 1) = (R + r * std::cos(theta)) * std::cos(phi) * dphi_dv;
148 gradient(1, 2) = 0.0;
149
150 return gradient;
151 }
152
153 double R;
154 double r;
155 int resolution;
156 double angle_offset;
157};

159template <typename PositionFunction>
160void generate_point_grid(
161 int resolution,
162 PositionFunction position_func,
163 std::vector< std::vector<VectorXr> > &point_grid
164) {
165 point_grid.resize(resolution);
166 for (int i = 0; i < resolution; ++i) {
167 point_grid[i].resize(resolution);
168 for (int j = 0; j < resolution; ++j) {
169 double u = i * 1.0;
170 double v = j * 1.0;
171 point_grid[i][j] = position_func(u, v);
172 }
173 }

221template <typename GradientFunction>
222void generate_parametric_affine_manifold_edge_gradients(
223 GradientFunction &gradient_func,
224 ParametricAffineManifold &parametric_affine_manifold,
225 Halfedge &halfedge,
226 std::vector<std::array<SpatialVector, 3>> &edge_gradients
227) {
228 Eigen::MatrixXi const & F = parametric_affine_manifold.get_faces();
229 size_t num_faces = parametric_affine_manifold.num_faces();
230
231 edge_gradients.resize(num_faces);
232 for (size_t face_index = 0; face_index < num_faces; ++face_index)
233 {
234 for (size_t face_vertex_index = 0; face_vertex_index < 3; ++face_vertex_index)
235 {
236 EdgeManifoldChart const & chart = parametric_affine_manifold.get_edge_chart(
237 face_index,
238 face_vertex_index
239 );
240 size_t f_top = chart.top_face_index;
241 size_t v_top = chart.top_vertex_index;
242 int j_top = find_face_vertex_index(F.row(f_top), v_top);
243 if ( f_top != face_index ) continue; // Only process top faces of edge charts to prevent redundancy
244 assert( j_top == face_vertex_index );
245
246 // Get face indices
247 size_t i = F(face_index, (face_vertex_index + 1) % 3);
248 size_t j = F(face_index, (face_vertex_index + 2) % 3);
249 size_t k = F(face_index, face_vertex_index);
250 assert (k == v_top);
251
252 // Get vertex uv positions
253 PlanarPoint uvi, uvj;
254 parametric_affine_manifold.get_vertex_global_uv(i, uvi);
255 parametric_affine_manifold.get_vertex_global_uv(j, uvj);
256
257 // Get midpoint uv position and gradient
258 PlanarPoint uvij = 0.5 * (uvi + uvj);
259 Matrix2x3r Gij = gradient_func(uvij[0], uvij[1]);
260 spdlog::debug(
261 "Gradient for corner {}, {} is: \n{}",
262 face_index,
263 face_vertex_index,
264 Gij
265 );
266
267 // Get normal uv directions perpendicular to the edge direction
268 Matrix2x2r R;
269 R <<
270 0, 1,
271 -1, 0;
272 PlanarPoint dji = (uvj - uvi);
273 PlanarPoint uv_perp_ij = dji * R;
274
275 // Build midpoint normals (indexed opposite the edge)
276 edge_gradients[face_index][face_vertex_index] = uv_perp_ij * Gij;
277 spdlog::debug(
278 "Reduced edge gradient for corner {}, {} is: \n{}",
279 face_index,
280 face_vertex_index,
281 edge_gradients[face_index][face_vertex_index]
282 );
283
284 // If the edge isn't on the boundary, set the other face corner corresponding to it
285 if (!chart.is_boundary)
286 {
287 size_t f_bottom = chart.bottom_face_index;
288 size_t v_bottom = chart.bottom_vertex_index;
289 int j_bottom = find_face_vertex_index(F.row(f_bottom), v_bottom);
290 edge_gradients[f_bottom][j_bottom] = edge_gradients[face_index][face_vertex_index];
291 }
292 }
293 }
294}

297template <typename PositionFunction, typename GradientFunction>
298void generate_parametric_affine_manifold_corner_data(
299 PositionFunction position_func,
300 GradientFunction gradient_func,
301 ParametricAffineManifold &parametric_affine_manifold,
302 std::vector<std::array<TriangleCornerFunctionData, 3>> &corner_data
303) {
304 size_t num_faces = parametric_affine_manifold.num_faces();
305 corner_data.resize(num_faces);
306 for (size_t face_index = 0; face_index < num_faces; ++face_index)
307 {
308 // Get face vertex indices
309 Eigen::MatrixXi const & F = parametric_affine_manifold.get_faces();
310 size_t i = F(face_index, 0);
311 size_t j = F(face_index, 1);
312 size_t k = F(face_index, 2);
313
314 // Get vertex uv positions
315 PlanarPoint uvi, uvj, uvk;
316 parametric_affine_manifold.get_vertex_global_uv(i, uvi);
317 parametric_affine_manifold.get_vertex_global_uv(j, uvj);
318 parametric_affine_manifold.get_vertex_global_uv(k, uvk);
319
320 // Get vertex positions
321 SpatialVector vi = position_func(uvi[0], uvi[1]);
322 SpatialVector vj = position_func(uvj[0], uvj[1]);
323 SpatialVector vk = position_func(uvk[0], uvk[1]);
324
325 // Get vertex gradients
326 Matrix2x3r Gi = gradient_func(uvi[0], uvi[1]);
327 Matrix2x3r Gj = gradient_func(uvj[0], uvj[1]);
328 Matrix2x3r Gk = gradient_func(uvk[0], uvk[1]);
329
330 // Get uv directions
331 PlanarPoint dij = uvj - uvi;
332 PlanarPoint dik = uvk - uvi;
333 PlanarPoint djk = uvk - uvj;
334 PlanarPoint dji = uvi - uvj;
335 PlanarPoint dki = uvi - uvk;
336 PlanarPoint dkj = uvj - uvk;
337
338 // Build first corner data
339 corner_data[face_index][0].function_value = vi;
340 corner_data[face_index][0].first_edge_derivative = dij * Gi;
341 corner_data[face_index][0].second_edge_derivative = dik * Gi;
342
343 // Build second corner data
344 corner_data[face_index][1].function_value = vj;
345 corner_data[face_index][1].first_edge_derivative = djk * Gj;
346 corner_data[face_index][1].second_edge_derivative = dji * Gj;
347
348 // Build third corner data
349 corner_data[face_index][2].function_value = vk;
350 corner_data[face_index][2].first_edge_derivative = dki * Gk;
351 corner_data[face_index][2].second_edge_derivative = dkj * Gk;
352 }
353}
