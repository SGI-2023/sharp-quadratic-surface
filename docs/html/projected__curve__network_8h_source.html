105 // Split segment into two new segments at a given knot value
106 void split_at_knot(double knot,
107 SegmentGeometry& lower_segment,
108 SegmentGeometry& upper_segment)
109 {
110 // Split all three segment curves at the knot
111 Conic lower_parameter_curve;
112 RationalFunction<4, 3> lower_spatial_curve;
113 RationalFunction<4, 2> lower_planar_curve;
114 Conic upper_parameter_curve;
115 RationalFunction<4, 3> upper_spatial_curve;
116 RationalFunction<4, 2> upper_planar_curve;
117 m_parameter_curve.split_at_knot(
118 knot, lower_parameter_curve, upper_parameter_curve);
119 m_planar_curve.split_at_knot(knot, lower_planar_curve, upper_planar_curve);
120 m_spatial_curve.split_at_knot(
121 knot, lower_spatial_curve, upper_spatial_curve);
122
123 // Build new segments from the split curves
124 lower_segment = SegmentGeometry(lower_parameter_curve,
125 lower_spatial_curve,
126 lower_planar_curve,
127 m_segment_labels);
128 upper_segment = SegmentGeometry(upper_parameter_curve,
129 upper_spatial_curve,
130 upper_planar_curve,
131 m_segment_labels);
132 }
133
134private:
135 RationalFunction<4, 2> m_planar_curve;
136 RationalFunction<4, 3> m_spatial_curve;
137 Conic m_parameter_curve;
138 std::map<std::string, int> m_segment_labels;
139 int m_quantitative_invisibility;
140
141 bool is_valid_segment_geometry() const
142 {
143 double t0 = m_planar_curve.domain().get_lower_bound();
144 double t1 = m_planar_curve.domain().get_upper_bound();
145 if (m_spatial_curve.domain().get_lower_bound() != t0) {
146 spdlog::error("Lower bound error");
147 return false;
148 }
149 if (m_parameter_curve.domain().get_lower_bound() != t0) {
150 spdlog::error("Lower bound error");
151 return false;
152 }
153 if (m_spatial_curve.domain().get_upper_bound() != t1) {
154 spdlog::error("Upper bound error");
155 return false;
156 }
157 if (m_parameter_curve.domain().get_upper_bound() != t1) {
158 spdlog::error("Upper bound error");
159 return false;
160 }
161
162 return true;
163 }

423 class SegmentChainIterator
424 {
425 public:
426 SegmentChainIterator(const ProjectedCurveNetwork& parent,
427 SegmentIndex segment_index)
428 : m_parent(parent)
429 {
430 m_current_segment_index = segment_index;
431 m_is_end_of_chain = false;
432 m_is_reverse_end_of_chain = false;
433 spdlog::trace("Iterator initialized to segment {}",
434 m_current_segment_index);
435 }
436
437 SegmentChainIterator& operator++()
438 {
439 // Check if off end of chain
440 if (m_is_end_of_chain || (m_is_reverse_end_of_chain))
441 return *this;
442
443 // Check if next is end of chain (i.e. the next node is not a knot node)
444 NodeIndex to_node = m_parent.to(m_current_segment_index);
445 if (!m_parent.is_knot_node(to_node))
446 // if
447 // (!m_parent.is_valid_segment_index(m_parent.next(m_current_segment_index)))
448 {
449 m_current_segment_index = -1;
450 m_is_end_of_chain = true;
451 } else {
452 m_current_segment_index = m_parent.next(m_current_segment_index);
453 }
454
455 spdlog::trace("Iterator moved to segment {}", m_current_segment_index);
456
457 return *this;
458 }
459
460 SegmentChainIterator operator++(int)
461 {
462 SegmentChainIterator temp = *this;
463 ++*this;
464 return temp;
465 }
466
467 SegmentChainIterator& operator--()
468 {
469 // Check if off end of chain
470 if (m_is_end_of_chain || (m_is_reverse_end_of_chain))
471 return *this;
472
473 // Check if prev is reverse end of chain (i.e. the prev node is not a knot
474 // node)
475 NodeIndex from_node = m_parent.from(m_current_segment_index);
476 if (!m_parent.is_knot_node(from_node))
477 // if
478 // (!m_parent.is_valid_segment_index(m_parent.prev(m_current_segment_index)))
479 {
480 m_current_segment_index = -1;
481 m_is_reverse_end_of_chain = true;
482 } else {
483 m_current_segment_index = m_parent.prev(m_current_segment_index);
484 }
485
486 spdlog::trace("Iterator moved to segment {}", m_current_segment_index);
487
488 return *this;
489 }
490
491 SegmentChainIterator operator--(int)
492 {
493 SegmentChainIterator temp = *this;
494 --*this;
495 return temp;
496 }
497
498 bool at_end_of_chain() { return m_is_end_of_chain; }
499
500 bool at_reverse_end_of_chain() { return m_is_reverse_end_of_chain; }
501
502 SegmentIndex operator*() { return m_current_segment_index; }
503
504 private:
505 const ProjectedCurveNetwork& m_parent;
506 SegmentIndex m_current_segment_index;
507 bool m_is_end_of_chain;
508 bool m_is_reverse_end_of_chain;
509 };
