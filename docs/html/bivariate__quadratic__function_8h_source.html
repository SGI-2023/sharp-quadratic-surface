186template<int dimension>
187std::string
188formatted_bivariate_quadratic_mapping(
189 const Eigen::Ref<const Eigen::Matrix<double, 6, dimension>>& quadratic_coeffs,
190 size_t precision = 16)
191{
192 std::stringstream quadratic_string;
193 for (int i = 0; i < quadratic_coeffs.cols(); ++i) {
194 quadratic_string << std::fixed << std::setprecision(precision)
195 << quadratic_coeffs(0, i);
196 quadratic_string << formatted_term(quadratic_coeffs(1, i), "u", precision);
197 quadratic_string << formatted_term(quadratic_coeffs(2, i), "v", precision);
198 quadratic_string << formatted_term(quadratic_coeffs(3, i), "uv", precision);
199 quadratic_string << formatted_term(
200 quadratic_coeffs(4, i), "u^2", precision);
201 quadratic_string << formatted_term(
202 quadratic_coeffs(5, i), "v^2", precision);
203 quadratic_string << "\n";
204 }
205
206 return quadratic_string.str();
207}

240template<typename Scalar>
241void
242generate_quadratic_coordinate_affine_transformation_matrix(
243 const Eigen::Matrix<Scalar, 2, 2>& linear_transformation,
244 const Eigen::Matrix<Scalar, 1, 2>& translation,
245 Eigen::Matrix<Scalar, 6, 6>& change_of_basis_matrix)
246{
247 // Get matrix information
248 Scalar b11 = linear_transformation(0, 0);
249 Scalar b12 = linear_transformation(0, 1);
250 Scalar b21 = linear_transformation(1, 0);
251 Scalar b22 = linear_transformation(1, 1);
252 Scalar b1 = translation[0];
253 Scalar b2 = translation[1];
254 Scalar one(1);
255 Scalar zero(0);
256
257 // Set matrix
258 change_of_basis_matrix << one, b1, b2, b1 * b2, b1 * b1, b2 * b2, zero, b11,
259 b12, b11 * b2 + b12 * b1, 2 * b1 * b11, 2 * b2 * b12, zero, b21, b22,
260 b22 * b1 + b21 * b2, 2 * b1 * b21, 2 * b2 * b22, zero, zero, zero,
261 b11 * b22 + b21 * b12, 2 * b11 * b21, 2 * b12 * b22, zero, zero, zero,
262 b11 * b12, b11 * b11, b12 * b12, zero, zero, zero, b21 * b22, b21 * b21,
263 b22 * b22;
264}

273template<typename Scalar>
274void
275generate_quadratic_coordinate_translation_matrix(
276 double du,
277 double dv,
278 Eigen::Matrix<Scalar, 6, 6>& change_of_basis_matrix)
279{
280 // Build (inverse) translation and linear identity matrix
281 Eigen::Matrix<Scalar, 2, 1> translation = { -du, -dv };
282 Eigen::Matrix<Scalar, 2, 2> identity;
283 identity << Scalar(1), Scalar(0), Scalar(0), Scalar(1);
284
285 // Generate the translation transformation as a special case of an affine
286 // transformation
287 generate_quadratic_coordinate_affine_transformation_matrix<Scalar>(
288 identity, translation, change_of_basis_matrix);
289}

300template<typename Scalar>
301void
302generate_quadratic_coordinate_barycentric_transformation_matrix(
303 const Eigen::Matrix<Scalar, 3, 3>& barycentric_transformation,
304 Eigen::Matrix<Scalar, 6, 6>& change_of_basis_matrix)
305{
306 Eigen::Matrix<Scalar, 2, 2> linear_transformation;
307 Eigen::Matrix<Scalar, 2, 1> translation;
308
309 // Get relevant barycentric matrix components. Since the w coordinate is
310 // discarded, we only use the first two rows of the barycentric transformation
311 Scalar a11 = barycentric_transformation(0, 0);
312 Scalar a12 = barycentric_transformation(0, 1);
313 Scalar a13 = barycentric_transformation(0, 2);
314 Scalar a21 = barycentric_transformation(1, 0);
315 Scalar a22 = barycentric_transformation(1, 1);
316 Scalar a23 = barycentric_transformation(1, 2);
317
318 // Build affine transformation from barycentric transformation with w = 1 - u
319 // - v
320 linear_transformation << a11 - a13, a21 - a23, a12 - a13, a22 - a23;
321 translation << a13, a23;
322
323 // Get change of basis matrix from the affine transformation
324 generate_quadratic_coordinate_affine_transformation_matrix<Scalar>(
325 linear_transformation, translation, change_of_basis_matrix);
326}

338template<typename Scalar>
339void
340generate_quadratic_coordinate_domain_triangle_normalization_matrix(
341 const Eigen::Matrix<Scalar, 2, 1>& v0,
342 const Eigen::Matrix<Scalar, 2, 1>& v1,
343 const Eigen::Matrix<Scalar, 2, 1>& v2,
344 Eigen::Matrix<Scalar, 6, 6>& change_of_basis_matrix)
345{
346 // Generate affine transformation mapping the standard triangle to the new
347 // triangle
348 Eigen::Matrix<Scalar, 2, 2> linear_transformation;
349 Eigen::Matrix<Scalar, 2, 1> translation;
350 linear_transformation.row(0) = v1 - v0;
351 linear_transformation.row(1) = v2 - v0;
352 translation = v0;
353
354 // Get change of basis matrix from the affine transformation
355 generate_quadratic_coordinate_affine_transformation_matrix<Scalar>(
356 linear_transformation, translation, change_of_basis_matrix);
357}

365template<typename Scalar>
366void
367generate_reparametrization(
368 const Eigen::Matrix<Scalar, 2, 1>& v0,
369 const Eigen::Matrix<Scalar, 2, 1>& v1,
370 const Eigen::Matrix<Scalar, 2, 1>& v2,
371 Eigen::Matrix<Scalar, 2, 2>& general_to_canonical_reparametrization,
372 Eigen::Matrix<Scalar, 2, 2>& canonical_to_general_reparametrization)
373{
374 Scalar l0 = (v1 - v0).norm();
375 ;
376 Scalar l1 = (v2 - v1).norm();
377 ;
378 Scalar l2 = (v0 - v2).norm();
379 ;
380 Scalar cos_angle = (l0 * l0 + l2 * l2 - l1 * l1) / (2 * l0 * l2);
381 Scalar sin_angle = sqrt(1 - cos_angle * cos_angle);
382 general_to_canonical_reparametrization << Scalar(1) / l0, Scalar(0),
383 -cos_angle / (l0 * sin_angle), Scalar(1) / (l2 * sin_angle);
384 canonical_to_general_reparametrization << l0, l2 * cos_angle, Scalar(0),
385 l2 * sin_angle;
386}
